\chapter{Introduction}\label{chap:intro}

A famous chemist P.~Flory proposed to consider self-avoiding (\emph{i.e.}\ visiting
every vertex at most once) walks on a lattice as a model for polymer chains.
Denote by $c_n$ the number of $n$-step self-avoiding walks on the hexagonal lattice
$\mathbb{H}$ started from some fixed vertex. Since a $(n{+}m)$-step self-avoiding
walk can be uniquely cut into an $n$-step and (a translation of) an $m$-step walk,
$c_{n+m} \leq c_n c_m$, from which it follows that
$\mu := \lim_{n \to \infty} c_n^{1/n}$ exists and is positive.
The positive real number $\mu$ is called the \emph{connective constant} of the
hexagonal lattice.

\begin{lemma}[Boundary cosine positivity]\label{lem:boundary_cos_pos}
\lean{boundary_cos_pos}
\leanok

  $\cos(3\theta/8) > 0$.

\end{lemma}

\begin{lemma}[Strip identity, existential form]\label{lem:strip_identity_exists}
\lean{strip_identity_exists}
\uses{thm:vertex_relation, lem:boundary_cos_pos, lem:left_coeff, lem:right_coeff}
  There exist non-negative reals $A_m, E_m$ (mid-edge-based partition
  functions) such that
  \[ 1 = c_\alpha A_m + B_{\text{paper}}(T,L,x_c) + c_\varepsilon E_m. \]
  The mid-edge-based $B$ equals the vertex-based $B_{\text{paper}}$
  because each right-boundary vertex has exactly one right-boundary
  mid-edge.

  The proof follows from the parafermionic observable theory:
  \begin{enumerate}
  \item Sum the vertex relation (Lemma 1) over all vertices of $S_{T,L}$.
  \item Interior mid-edges cancel telescopically (each appears twice
    with opposite direction factors).
  \item Only boundary mid-edges survive; the boundary sum equals $0$.
  \item The starting mid-edge contributes $-1/2$ (real).
  \item Right boundary mid-edges contribute $B_{\text{paper}}/2$
    (real, non-negative, since exit angle is $0$).
  \item All other boundary contributions have non-negative real parts
    because $\cos(3\theta/8) > 0$ for all hex lattice exit angles $\theta$
    (by \texttt{boundary\_cos\_pos}).
  \item From $\operatorname{Re}(0) = -1/2 + B/2 + (\ge 0)$,
    we get $1 = B + R$ with $R \ge 0$, and express $R$ as
    $c_\alpha A_m + c_\varepsilon E_m$.
  \end{enumerate}

  Key algebraic facts (already proved):
  \begin{itemize}
  \item \texttt{pair\_cancellation}: $j \cdot \bar\lambda^4 + \bar{j} \cdot \lambda^4 = 0$
  \item \texttt{triplet\_cancellation}: $1 + x_c j \bar\lambda + x_c \bar{j} \lambda = 0$
  \end{itemize}

  Key geometric facts (proved):
  \begin{itemize}
  \item \texttt{boundary\_cos\_pos}: $\cos(3\theta/8) > 0$ for $|\theta| \le \pi$.
  \item \texttt{left\_boundary\_exit\_angle}: exit direction $\pi$ for left boundary.
  \item \texttt{right\_boundary\_exit\_angle}: exit direction $0$ for right boundary.
  \item \texttt{cos\_five\_pi\_eight}: $\cos(5\pi/8) = -c_\alpha$.
  \end{itemize}
\end{lemma}
\begin{proof}
  The full proof requires formalizing the parafermionic observable,
  the discrete Stokes theorem (interior cancellation), and the
  boundary winding evaluation. The algebraic core (pair and triplet
  cancellation) is proved. The geometric core (boundary cosine
  positivity) is proved. The combinatorial core (walk pairing and
  tripling bijection, interior cancellation) remains to be formalized.
\end{proof}

\begin{lemma}[Finiteness of paper strip]\label{lem:paper_fin_strip_finite}

\begin{definition}[Bridge decomposition]\label{def:bridge_decomp}
\lean{BridgeDecomposition, FullBridgeDecomposition}
\leanok
\uses{def:bridge, def:half_plane}
  Any self-avoiding walk can be canonically decomposed into a sequence
  of bridges with monotonically changing widths. Fixing the starting
  mid-edge and first vertex, the decomposition uniquely determines the walk.
  This is the Hammersley--Welsh decomposition.

  The decomposition algorithm:
  \begin{enumerate}
    \item For a half-plane walk: find the last vertex with maximal $x$-coordinate
      (step $n$). The first $n$ steps form a bridge of width $T_0$.
      The remaining steps form a half-plane walk of width $T_1 < T_0$. Recurse.
    \item For a general walk: split at the first vertex with maximal $x$-coordinate.
      The left part is a reverse half-plane walk, the right part a half-plane walk.
      Decompose each part separately.
  \end{enumerate}
\end{definition}

  \textbf{Status: sorry.} This is the key remaining combinatorial gap.
  The proof requires formalizing:
  \begin{enumerate}
  \item \textbf{Half-plane walk decomposition} (by strong induction on width $T_0$):
    \begin{itemize}
    \item Find the last vertex with maximal $x$-coordinate, say at step $n$.
    \item The first $n$ steps form a bridge $\tilde{\gamma}_1$ of width $T_0$.
    \item Forget the $(n{+}1)$-th vertex (its position is unambiguous).
    \item The remaining steps form a half-plane walk $\tilde{\gamma}_2$ of width $T_1 < T_0$.
    \item By the induction hypothesis, $\tilde{\gamma}_2$ decomposes into bridges of widths $T_1 > \cdots > T_j$.

    \begin{itemize}
    \item Find the first vertex with maximal $x$-coordinate, splitting $\gamma$ into $\gamma_1$ and $\gamma_2$.
    \item $\gamma_1$ is a reverse half-plane walk; $\gamma_2$ is a half-plane walk.
    \item Decompose each part, giving widths $T_{-i} < \cdots < T_{-1}$ and $T_0 > \cdots > T_j$.

  \item \textbf{Injectivity}: The reverse procedure uniquely reconstructs the walk from the bridge

  \item \textbf{Weight accounting}: Walk length $n \geq$ sum of bridge lengths (due to forgotten
    vertices between consecutive bridges). Since $0 < x \leq 1$ (as $x < x_c < 1$),
    $x^n \leq \prod x^{\ell(\beta_i)}$.
  \end{enumerate}

\chapter{Conjectures}\label{chap:conjectures}

The following conjectures are stated in Section~4 of the paper.
They concern the precise asymptotic behavior of self-avoiding walks
and their conjectured conformal invariance in the scaling limit.

\begin{definition}[Critical exponents]\label{def:exponents}
\lean{gamma_SAW, nu_SAW}
\leanok
\uses{def:connective_constant}
  The conjectured critical exponents: $\gamma = 43/32$ and $\nu = 3/4$.
\end{definition}

\begin{conjecture}[Nienhuis asymptotic formula]\label{conj:nienhuis}
\lean{nienhuis_asymptotic_conjecture}
\leanok
\uses{def:exponents, def:connective_constant}
  $c_n \sim A \cdot n^{\gamma-1} \cdot \mu^n$ with $\gamma = 43/32$.
\end{conjecture}

\lean{flory_exponent_conjecture}
\leanok

\end{conjecture}
