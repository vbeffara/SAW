\chapter{Introduction}\label{chap:intro}

A famous chemist P.~Flory proposed to consider self-avoiding (\emph{i.e.}\ visiting
every vertex at most once) walks on a lattice as a model for polymer chains.
Denote by $c_n$ the number of $n$-step self-avoiding walks on the hexagonal lattice
$\mathbb{H}$ started from some fixed vertex. Since a $(n{+}m)$-step self-avoiding
walk can be uniquely cut into an $n$-step and (a translation of) an $m$-step walk,
$c_{n+m} \leq c_n c_m$, from which it follows that
$\mu := \lim_{n \to \infty} c_n^{1/n}$ exists and is positive.
The positive real number $\mu$ is called the \emph{connective constant} of the
hexagonal lattice.

\begin{lemma}[Boundary cosine positivity]\label{lem:boundary_cos_pos}

\begin{lemma}[Strip identity, existential form]\label{lem:strip_identity_exists}
\lean{strip_identity_exists}
\uses{thm:vertex_relation, lem:boundary_cos_pos, lem:left_coeff, lem:right_coeff}
  There exist non-negative reals $A_m, E_m$ (mid-edge-based partition
  functions) such that
  \[ 1 = c_\alpha A_m + B_{\text{paper}}(T,L,x_c) + c_\varepsilon E_m. \]
  The mid-edge-based $B$ equals the vertex-based $B_{\text{paper}}$
  because each right-boundary vertex has exactly one right-boundary
  mid-edge.

  The proof follows from the parafermionic observable theory:
  \begin{enumerate}
  \item Sum the vertex relation (Lemma 1) over all vertices of $S_{T,L}$.
  \item Interior mid-edges cancel telescopically (each appears twice
    with opposite direction factors).
  \item Only boundary mid-edges survive; the boundary sum equals $0$.
  \item The starting mid-edge contributes $-1/2$ (real).
  \item Right boundary mid-edges contribute $B_{\text{paper}}/2$
    (real, non-negative, since exit angle is $0$).
  \item All other boundary contributions have non-negative real parts
    because $\cos(3\theta/8) > 0$ for all hex lattice exit angles $\theta$
    (by \texttt{boundary\_cos\_pos}).
  \item From $\operatorname{Re}(0) = -1/2 + B/2 + (\ge 0)$,
    we get $1 = B + R$ with $R \ge 0$, and express $R$ as
    $c_\alpha A_m + c_\varepsilon E_m$.
  \end{enumerate}

  Key algebraic facts (already proved):
  \begin{itemize}
  \item \texttt{pair\_cancellation}: $j \cdot \bar\lambda^4 + \bar{j} \cdot \lambda^4 = 0$
  \item \texttt{triplet\_cancellation}: $1 + x_c j \bar\lambda + x_c \bar{j} \lambda = 0$
  \end{itemize}

  Key geometric facts (proved):
  \begin{itemize}
  \item \texttt{boundary\_cos\_pos}: $\cos(3\theta/8) > 0$ for $|\theta| \le \pi$.
  \item \texttt{left\_boundary\_exit\_angle}: exit direction $\pi$ for left boundary.
  \item \texttt{right\_boundary\_exit\_angle}: exit direction $0$ for right boundary.
  \item \texttt{cos\_five\_pi\_eight}: $\cos(5\pi/8) = -c_\alpha$.
  \end{itemize}
\end{lemma}
\begin{proof}
  \textbf{Status: sorry.}
  The full proof requires formalizing the parafermionic observable,
  the discrete Stokes theorem (interior cancellation), and the
  boundary winding evaluation.

  \textbf{What is proved:}
  \begin{itemize}

\begin{remark}[Coordinate system for strips]\label{rem:coord_system}
  \textbf{Important correction.}
  The paper defines the strip $S_T$ using the real part of the embedding, which
  corresponds to the \emph{diagonal coordinate} $x + y$ in our $(x, y, \mathrm{bool})$
  representation. The original Lean definition \texttt{Bridge} uses the
  \emph{column coordinate} $x$ instead. This mismatch is significant: the
  strip identity $B_T^{x_c} \leq 1$ holds for diagonal strips but not for
  column strips (the column-strip partition function can exceed $1$).

  The corrected formalization uses \texttt{DiagBridge} (SAWDiagBridge.lean),
  which defines bridges using the diagonal coordinate $x + y$, matching the paper.
  The key property is that each hex step changes $x + y$ by at most $1$
  (\texttt{hexGraph\_adj\_sum\_bound}), so diagonal bridges of width $T$
  have length $\geq T$ (\texttt{diag\_bridge\_length\_ge\_width}).
\end{remark}

\begin{definition}[Bridge decomposition]\label{def:bridge_decomp}
\lean{BridgeDecomposition, FullBridgeDecomposition}
\leanok
\uses{def:bridge, def:half_plane}
  Any self-avoiding walk can be canonically decomposed into a sequence
  of bridges with monotonically changing widths. Fixing the starting
  mid-edge and first vertex, the decomposition uniquely determines the walk.
  This is the Hammersley--Welsh decomposition.

  The decomposition algorithm:
  \begin{enumerate}
    \item For a half-plane walk: find the last vertex with maximal $x$-coordinate
      (step $n$). The first $n$ steps form a bridge of width $T_0$.
      The remaining steps form a half-plane walk of width $T_1 < T_0$. Recurse.
    \item For a general walk: split at the first vertex with maximal $x$-coordinate.
      The left part is a reverse half-plane walk, the right part a half-plane walk.
      Decompose each part separately.
  \end{enumerate}
\end{definition}

\subsection{Remaining sorry gaps}

The following sorry-containing declarations are the key remaining gaps
in the complete formalization.

\paragraph{Strip identity (Lemma~2 of the paper).}
The concrete strip identity has been \textbf{reduced} to a single key bound:
$B_{\mathrm{paper}}(T,L,x_c) \leq 1$ (\texttt{B\_paper\_le\_one\_direct},
SAWStripIdentityCorrect.lean). The reduction is proved as
\texttt{strip\_identity\_from\_B\_bound}: given $B_{\mathrm{paper}} \leq 1$,
set $A_m = (1 - B_{\mathrm{paper}})/c_\alpha$ and $E_m = 0$ to obtain
$1 = c_\alpha A_m + B_{\mathrm{paper}} + c_\varepsilon E_m$.

Proving $B_{\mathrm{paper}} \leq 1$ requires formalizing the summation of
the vertex relation over all vertices in the finite strip $S_{T,L}$,
together with interior cancellation and boundary winding analysis.
The algebraic core (pair/triplet cancellation) and boundary coefficients
($c_\alpha$, $c_\varepsilon$) are fully proved.

\paragraph{Column vs.\ diagonal coordinate mismatch.}
\textbf{Important:} The original \texttt{Bridge} definition (SAWBridgeFix.lean)
uses \emph{column} coordinates ($x$-coordinate in $[0,T]$), while the paper's
strip identity uses \emph{diagonal} coordinates ($x+y$ in $[0,T]$). The
boundary geometry differs between the two coordinate systems, so the strip
identity $B_{\mathrm{paper}} \leq 1$ does \emph{not} directly imply
\texttt{origin\_bridge\_upper\_bound} for column bridges.
The correct formalization uses \texttt{PaperSAW\_B} (from
SAWStripIdentityCorrect.lean), which matches the paper's diagonal domain
with sublattice-dependent boundary conditions.
A hex lattice reflection automorphism $(x,y,b) \mapsto (-y,-x,\lnot b)$
(proved in SAWDiagConnection.lean as \texttt{hexReflect\_adj}) relates
the paper's negative-diagonal strip to positive-diagonal bridges.

\paragraph{Bridge partition bounds (consequences of strip identity).}
The following bounds are all \emph{proved modulo the strip identity}:
\begin{itemize}
  \item \texttt{origin\_bridge\_partial\_sum\_le\_one} (SAWFiniteStrip.lean):
    For any finite set $F$ of origin bridges, $\sum_{b \in F} x_c^{\ell(b)} \leq 1$.
    \textbf{Proved} from \texttt{B\_TL\_le\_one} via the injection
    \texttt{originBridgeToStripB}.
  \item \texttt{origin\_bridge\_upper\_bound\_from\_strip} (SAWFiniteStrip.lean):
    $B_T^{x_c} \leq 1$. \textbf{Proved} from partial sum bounds via
    \texttt{Real.tsum\_le\_of\_sum\_le}.
  \item \texttt{origin\_bridge\_upper\_bound} (SAWBridgeFix.lean):
    Same statement, still sorry (earlier version; superseded by the above).
  \item \texttt{origin\_bridge\_lower\_bound} (SAWBridgeFix.lean):
    $\exists c > 0,\, B_T^{x_c} \geq c/T$. Still sorry. Would follow from the
    strip identity combined with the quadratic recurrence (proved in SAWDecomp.lean).
  \item \texttt{B\_T\_inf\_eq\_origin\_bridge} (SAWFiniteStrip.lean):
    Connecting $\sup_L B_{T,L}$ to \texttt{origin\_bridge\_partition}. Still sorry.
\end{itemize}

\paragraph{Proved infrastructure.}
\begin{itemize}
  \item \texttt{finite\_strip\_finite}: The finite strip vertex set is finite.
  \item \texttt{strip\_saw\_length\_bound}: SAW length in strip $\leq$ number of strip vertices.
  \item \texttt{B\_TL\_summable}: Partition functions on finite strips are summable
    (\texttt{StripSAW\_B T L} is a finite type).
  \item \texttt{originBridgeToStripB\_injective}: The origin-bridge-to-strip map is injective.
  \item \texttt{originBridgeToStripB\_len}: The map preserves walk length.

  \item \texttt{zigzag\_endpoint\_true\_not\_in\_support}: Extension lemma

\paragraph{Bridge decomposition injection (Hammersley--Welsh).}
\texttt{bridge\_decomposition\_injection\_proof} (SAWHWDecomp.lean):
The core counting inequality from the Hammersley--Welsh bridge
decomposition:
$\sum_{n \leq N} c_n x^n \leq 2 \big(\sum_{S \subseteq \{0,\ldots,N-1\}}
\prod_{T \in S} B_{T+1}^x\big)^2$.
Requires formalizing the recursive half-plane walk decomposition into
bridges (by induction on width), splitting general SAWs at the first

accounting argument ($x^n \leq \prod x^{\ell_i}$ for $0 < x \leq 1$).

\section{Remaining gaps}

The main theorem and its complete proof chain compile without error in Lean.
All sorries ultimately reduce to two fundamental gaps.

\subsection{Gap~1: Strip identity (discrete Stokes theorem)}

\texttt{B\_paper\_le\_one\_direct} (Theorem~\ref{thm:B_paper_le_one_direct}):
Proving $B_{\text{paper}}(T, L, x_c) \leq 1$ requires formalizing the
discrete Stokes theorem:

\begin{enumerate}
  \item \textbf{Interior cancellation}: When summing the vertex relation
    $(p-v)F(p) + (q-v)F(q) + (r-v)F(r) = 0$ over all strip vertices,
    each interior mid-edge $z$ appears from exactly two vertices $v_1, v_2$
    with coefficients $(z-v_1) + (z-v_2) = 0$, since $z$ is the midpoint
    of the edge $(v_1, v_2)$.
  \item \textbf{Boundary evaluation}: Boundary mid-edges contribute once.
    The starting mid-edge $a$ contributes $-1/2$. Right boundary mid-edges
    contribute $B_{\text{paper}}/2$ (winding $= 0$ for the paper's diagonal
    strip). All other contributions have non-negative real parts by
    \texttt{boundary\_cos\_pos}.
  \item \textbf{Assembly}: $\operatorname{Re}(0) = -1/2 + B/2 + (\geq 0)$
    gives $B \leq 1$.
\end{enumerate}

The algebraic ingredients (pair/triplet cancellation) are fully proved.
The missing infrastructure is the combinatorial bookkeeping: defining
mid-edges of the strip, the interior/boundary classification, and the
formal telescoping argument.

Proving this unlocks both \texttt{origin\_bridge\_upper\_bound}
($B_T^{x_c} \leq 1$) and \texttt{origin\_bridge\_lower\_bound}
($B_T^{x_c} \geq c/T$) via the quadratic recurrence argument
(fully formalized in SAWQuadRecurrence.lean).

\subsection{Gap~2: Hammersley--Welsh bridge decomposition}

\texttt{bridge\_decomposition\_injection\_proof}
(Theorem~\ref{thm:bridge_decomp_inj_proof}):
Every self-avoiding walk decomposes into a unique sequence of bridges
with monotone widths. The proof requires:
\begin{enumerate}
  \item Half-plane walk decomposition by strong induction on width.
  \item General walk splitting at the first max-$x$ vertex.
  \item Injectivity of the reverse procedure.
  \item Weight accounting: walk length $\geq$ sum of bridge lengths.
\end{enumerate}

\subsection{Architectural issue: column vs.\ diagonal coordinates}

\textbf{Important:} The formalization uses two coordinate systems that are
not equivalent:
\begin{itemize}
  \item \textbf{Column bridges} (\texttt{OriginBridge} in SAWBridgeFix.lean):
    defined by the $x$-coordinate constraint $0 \leq x \leq T$.
  \item \textbf{Diagonal bridges} (\texttt{PaperSAW\_B} in
    SAWStripIdentityCorrect.lean): defined by the diagonal constraint
    $-T \leq x+y \leq 0$.
\end{itemize}

The strip identity $B_{\text{paper}} \leq 1$ holds for \emph{diagonal}
bridges because the right boundary mid-edges have winding $0$ from the
starting mid-edge (they are horizontal in the paper's embedding).
For column strips, the right boundary mid-edges have direction $2\pi/3$,
giving a weaker bound $B \leq 1/\cos(5\pi/12)$.

The proof chain (SAWHammersleyWelsh, SAWHWDecomp, SAWBridgeFix) uses
column bridges throughout. Closing Gap~1 therefore requires either:
\begin{enumerate}
  \item[(a)] Restructuring the proof chain to use diagonal bridges; or
  \item[(b)] Proving a column strip identity with different coefficients
    (which gives $B \leq C$ for some $C > 1$, then adjusting downstream
    constants); or
  \item[(c)] Finding a lattice automorphism relating column and diagonal
    bridges (no simple automorphism exists in the current coordinate system);
    or
  \item[(d)] Using \texttt{PaperBridge} (defined in SAWDiagProof.lean) which
    matches \texttt{PaperInfStrip} exactly. The bound
    $\texttt{paper\_bridge\_partition}(T, x_c) \leq 1/x_c$ follows from
    $B_{\text{paper}} \leq 1$ with the edge-count convention.
    The Hammersley--Welsh decomposition can be adapted to split at
    the vertex with minimal $x+y$ (instead of maximal $x$), producing
    diagonal bridges. The product convergence still works with the
    weaker bound $B \leq 1/x_c$ (any finite constant suffices).
\end{enumerate}

Option (d), using \texttt{PaperBridge} from SAWDiagProof.lean, is the
recommended approach. The definitions and key lemmas (bridge length $\geq T$,
fit in finite strip) are proved. The remaining sorry is
\texttt{paper\_bridge\_partial\_sum\_le}, which depends on
\texttt{B\_paper\_le\_one\_direct} (Gap~1).

\subsection{Superseded and non-critical sorries}

All other sorry'd declarations are either:

  \item Direct consequences of Gap~1 (e.g., \texttt{origin\_bridge\_upper\_bound},
    \texttt{origin\_bridge\_lower\_bound}, \texttt{origin\_bridge\_partial\_sum\_le\_one});
  \item Superseded by proved versions (e.g., \texttt{saw\_count\_even\_lower} is

\section{Bridge decomposition algorithm (Hammersley--Welsh)}\label{sec:hw_algorithm}

The bridge decomposition, first introduced by Hammersley and Welsh~\cite{HammersleyWelsh},
is the key combinatorial ingredient for the upper bound $\mu \leq \sqrt{2+\sqrt{2}}$.

\subsection{Half-plane walks}

A \emph{half-plane walk} of width~$T_0$ is a self-avoiding walk $\tilde{\gamma}$
such that the start of $\tilde{\gamma}$ has extremal diagonal coordinate.
Formally, for vertices using the diagonal coordinate $d(v) = v_1 + v_2$, the starting
vertex achieves $\min_{u \in \tilde{\gamma}} d(u)$.

\begin{definition}[Half-plane walk]\label{def:half_plane}
\uses{def:saw}
  A SAW $\gamma = (v_0, \ldots, v_n)$ is \emph{half-plane} if
  $d(v_0) \leq d(v_i)$ for all $i$. The \emph{width} is
  $T_0 = \max_i d(v_i) - d(v_0)$.
\end{definition}

\subsection{Recursive bridge extraction}

\begin{lemma}[Bridge extraction]\label{lem:bridge_extract}
\uses{def:half_plane, def:paper_bridge}
  Let $\tilde{\gamma}$ be a half-plane walk of width $T_0 \geq 1$.
  Among the vertices of $\tilde{\gamma}$ achieving $d(v) = d(v_0) + T_0$
  (the rightmost diagonal coordinate), let $v_k$ be the one visited last.
  Then:
  \begin{enumerate}
  \item The prefix $v_0, \ldots, v_k$ forms a bridge of width $T_0$.
  \item The vertex $v_{k+1}$ is uniquely determined by the bridge endpoint.
  \item The suffix $v_{k+2}, \ldots, v_n$ is a half-plane walk of width
        $T_1 < T_0$ (after forgetting $v_{k+1}$).
  \end{enumerate}
  \textbf{Status: not yet formalized.}
\end{lemma}

\subsection{Full walk decomposition}

\begin{lemma}[Split at deepest point]\label{lem:split_deepest}
\uses{def:saw}
  For any SAW $\gamma = (v_0, \ldots, v_n)$, let $v_k$ be the first vertex
  achieving $\min_i d(v_i)$. Then:
  \begin{enumerate}
  \item $\gamma_1 = (v_0, \ldots, v_k)$ reversed is a half-plane walk.
  \item $\gamma_2 = (v_k, \ldots, v_n)$ is a half-plane walk.
  \item The vertex sets of $\gamma_1$ and $\gamma_2$ share only $v_k$.
  \end{enumerate}
  \textbf{Status: not yet formalized.}
\end{lemma}

\begin{theorem}[Hammersley--Welsh decomposition]\label{thm:hw_decomp_algorithm}
\uses{lem:bridge_extract, lem:split_deepest}
  Any SAW $\gamma$ from the origin admits a canonical decomposition into
  two sequences of bridges with strictly monotone widths:
  $T_{-i} < \cdots < T_{-1}$ (from the first half) and
  $T_0 > \cdots > T_j$ (from the second half).

  The decomposition is injective: given the starting vertex and the
  sequences of bridge widths, the original walk can be uniquely
  reconstructed.
  \textbf{Status: not yet formalized.}
\end{theorem}

\subsection{Weight bound}

\begin{lemma}[Weight factorization]\label{lem:weight_factor}
\uses{thm:hw_decomp_algorithm}
  For $x \leq 1$, the walk weight $x^n$ satisfies
  $x^n \leq \prod_i x^{\ell(\beta_i)}$
  where $\beta_i$ are the bridges in the decomposition and $\ell(\beta_i)$
  is the edge-count of bridge $\beta_i$.
  This holds because the connecting vertices between bridges contribute
  additional factors of $x \leq 1$ that can only decrease the weight.
  \textbf{Status: not yet formalized.}
\end{lemma}

\subsection{Assembly of the injection bound}

\begin{theorem}[Bridge decomposition injection]\label{thm:hw_injection}
\lean{paper_bridge_decomp_injection}
\uses{thm:hw_decomp_algorithm, lem:weight_factor, def:paper_bridge_partition}
  For $0 < x < x_c$ and any $N$:
  $\sum_{n=0}^{N} c_n x^n \leq
    2 \cdot \left(\sum_{S \subseteq \{1,\ldots,N\}}
    \prod_{T \in S} B_T(x)\right)^2
    = 2 \cdot \prod_{T=1}^{N} (1 + B_T(x))^2$
  where $B_T(x) = \texttt{paper\_bridge\_partition}(T, x)$.

  The factor of~2 accounts for the two choices of starting direction.
  Each half gives a subset $S \subseteq \{1,\ldots,N\}$ (the bridge widths).
  By the Hammersley--Welsh decomposition, the map is injective, and by
  the weight factorization, each walk's contribution is bounded by the
  product of bridge weights.

  The identity $\sum_{S \subseteq F} \prod_{T \in S} f(T) = \prod_{T \in F} (1 + f(T))$
  is \texttt{Finset.prod\_one\_add} in Mathlib.
  \textbf{Status: sorry.}
\end{theorem}

\leanok

\leanok

  (by \texttt{extraWalk\_cut\_injective}) gives
  $\sum f(\varphi(s)) \leq \sum' f(p) = B^2$
  via \texttt{Summable.sum\_le\_tsum}.
\end{proof}

\lean{cutting_argument_proved}
\leanok

\end{lemma}

\leanok

  (bounded by $A_T$ via \texttt{in\_strip\_sum\_le}) and extra walks
  (bounded by $x_c B_{T+1}^2$ via \texttt{extra\_walk\_sum\_le\_proved}),
  then take the tsum limit.
\end{proof}

\begin{theorem}[Bridge recurrence from cutting]\label{thm:bridge_recurrence_from_cutting}
\lean{bridge_recurrence_from_cutting}
\leanok
\uses{lem:cutting, thm:strip_identity_genuine, thm:c_alpha_pos, thm:xc_pos}
  Assuming the cutting argument and the strip identity for the infinite strip,
  the bridge partition function satisfies the quadratic recurrence:
  $B_T \leq c_\alpha \cdot x_c \cdot B_{T+1}^2 + B_{T+1}$.
\end{theorem}
\begin{proof}
\leanok
\proves{thm:bridge_recurrence_from_cutting}
  Subtract the strip identity at $T$ and $T{+}1$:
  $0 = c_\alpha (A_{T+1} - A_T) + (B_{T+1} - B_T) + c_\varepsilon (E_{T+1} - E_T)$.
  Since $E_{T+1} \leq E_T$ (monotonicity), $B_T - B_{T+1} \leq c_\alpha (A_{T+1} - A_T)$.
  By the cutting argument, $A_{T+1} - A_T \leq x_c B_{T+1}^2$.
  Therefore $B_T \leq c_\alpha x_c B_{T+1}^2 + B_{T+1}$.
\end{proof}

\section{Vertex relation proof structure}\label{sec:vertex_rel_proof}

The proof of the vertex relation (Lemma~1) partitions walks ending at the
three mid-edges around a vertex $v$ into \emph{triplets} and \emph{pairs}.

\subsection{Walk classification}

For a walk $\gamma$ from $a$ to one of the three mid-edges $p, q, r$ around
a vertex $v$, define the number of mid-edges around $v$ \emph{visited} by
$\gamma$ (including start and end half-edges).

\begin{itemize}
\item \textbf{1 mid-edge}: The walk ends at $p$ and does not cross the other
  two edges around $v$. This occurs when:
  \begin{enumerate}
  \item[(a)] $v$ is the starting vertex and $\gamma$ has length~0, or
  \item[(b)] $\gamma$ enters $v$ from the same direction as the ending half-edge.
  \end{enumerate}

\item \textbf{2 mid-edges}: The walk enters $v$ from one direction and exits
  toward a different mid-edge. This is the generic case.

\item \textbf{3 mid-edges}: The walk uses all three edges around $v$. This
  requires the walk to enter $v$, exit to another neighbor, and the walk also
  crosses the third edge from the opposite side.
\end{itemize}

\subsection{Triplet grouping}

Walks visiting 1 mid-edge form triplets with walks visiting 2 mid-edges.
For each walk $\gamma_1$ visiting 1 mid-edge (ending at $p$):
\begin{itemize}
\item Extension $\gamma_2$: prolong $\gamma_1$ by one vertex step, ending at $q$.
  The walk now visits 2 mid-edges ($p$ and $q$).
\item Extension $\gamma_3$: prolong $\gamma_1$ by one vertex step, ending at $r$.
  The walk now visits 2 mid-edges ($p$ and $r$).
\end{itemize}
The contributions satisfy:
$c(\gamma_1) + c(\gamma_2) + c(\gamma_3) =
(p-v) \cdot e^{-i\sigma W_1} \cdot x_c^{\ell_1}
\cdot \underbrace{(1 + x_c \cdot j \cdot \bar{\lambda} + x_c \cdot \bar{j} \cdot \lambda)}_{=\,0} = 0$
by \texttt{triplet\_cancellation}.

\subsection{Pair grouping}

Walks visiting 3 mid-edges form pairs via loop reversal. If $\gamma_1$ enters
$v$ from $w_j$, exits to $w_l$, and the walk reaches $w_k$ from the opposite
side, then the partner $\gamma_2$ reverses the loop around $v$
(enters from $w_l$, exits to $w_j$, reaches $w_k$ from the opposite side).
The contributions satisfy:
$c(\gamma_1) + c(\gamma_2) =
(p-v) \cdot e^{-i\sigma W_{\text{common}}} \cdot x_c^{\ell}
\cdot \underbrace{(j \cdot \bar{\lambda}^4 + \bar{j} \cdot \lambda^4)}_{=\,0} = 0$
by \texttt{pair\_cancellation}.

\subsection{Exhaustiveness}

The grouping is exhaustive: every walk ending at one of the three mid-edges
around $v$ falls into exactly one triplet or one pair. The proof of exhaustiveness
requires showing that:
\begin{enumerate}
\item Every 1-mid-edge walk can be extended to both other mid-edges
  (the two unvisited neighbors are not on the walk's path).
\item Every 2-mid-edge walk is the extension of a unique 1-mid-edge walk.
\item Every 3-mid-edge walk has a unique loop-reversal partner.
\end{enumerate}
Point~(1) is the most delicate: if a neighbor $w_j$ of $v$ is already visited
by a walk ending at $v$, then the extension to $w_j$ does not exist, and the
triplet is incomplete. In this case, the walk should be grouped as a
3-mid-edge walk instead (the walk passes through $w_j$ via a path that
also uses edge $\{v, w_j\}$ from the opposite side).

\textbf{Status:} The algebraic identities (triplet and pair cancellation) are proved.
The phase shift lemmas (\texttt{exit\_phase\_w2}, \texttt{exit\_phase\_w3}) connecting
hex turns to $\lambda$ and $\bar\lambda$ are proved. All 18 hex turn values are
verified computationally (\texttt{SAWVertexRelCore2.lean}).
The \textbf{combinatorial grouping} (exhaustiveness of the partition into
pairs and triplets, including walk extension/contraction and loop reversal)
is not yet formalized.

retracting (removing the last step). These operations are formalized

\section{Hammersley--Welsh bridge decomposition}\label{sec:hw_decomp}

The Hammersley--Welsh bridge decomposition provides an injection from
SAWs (with a binary tag) to pairs of bridge sequences with strictly
decreasing widths. This is the content of Section~3 of the paper,
starting from ``any self-avoiding walk can be canonically decomposed
into a sequence of bridges''.

\begin{theorem}[Bridge decomposition injection]\label{thm:hw_decomp}
\lean{paper_bridge_decomp_injection}
\uses{lem:walkMaxDiagCoord_ge, lem:saw_suffix_half_plane, lem:bridge_weight}
  For $0 < x < x_c$ and any $N \in \mathbb{N}$,
  \[
    \sum_{n=0}^{N} c_n \, x^n
    \;\leq\;
    2 \left(
      \sum_{S \subseteq \{0,\ldots,N{-}1\}}
      \prod_{T \in S} B_{T+1}(x)
    \right)^{\!2}.
  \]
  \textbf{Status: sorry.}
  The proof requires:
  \begin{enumerate}
  \item Splitting each SAW at its first vertex of minimum diagCoord.
  \item Decomposing each half-plane walk into bridges of strictly
    decreasing widths (by strong induction on the walk width).
  \item Mapping each extracted bridge to a \texttt{PaperBridge}
    via a graph automorphism (translation + optional flip).
  \item Proving the decomposition map is injective.
  \item Verifying the weight factorization:
    $x^n = \prod x^{|b_i|}$ where $|b_i|$ are bridge lengths.
  \end{enumerate}
\end{theorem}

\begin{lemma}[$c_\alpha \cdot x_c = (\sqrt{2}-1)/2$]\label{lem:c_alpha_mul_xc}
\lean{c_alpha_mul_xc}
\leanok

\end{lemma}
\begin{proof}
\leanok

\section{Vertex relation (Lemma~1)}\label{sec:vertex_rel}

The vertex relation is the heart of the paper. It says that for each
interior vertex $v$ of a strip domain $\Omega$, the weighted sum of the
parafermionic observable over the three mid-edges adjacent to $v$ vanishes:
\begin{equation}\label{eq:vertex_rel}
  (p-v)F(p) + (q-v)F(q) + (r-v)F(r) = 0,
\end{equation}
where $p, q, r$ are the three mid-edges adjacent to $v$.

The proof partitions walks ending at $p$, $q$, or $r$ into:
\begin{itemize}
\item \textbf{Triplets}: a walk visiting one mid-edge, paired with two
  walks visiting two mid-edges (obtained by extending one step through $v$).
\item \textbf{Pairs}: two walks visiting all three mid-edges, differing
  by the direction of the ``loop'' around $v$.
\end{itemize}

The algebraic cancellation for triplets uses the identity
$1 + x_c j \overline{\lambda} + x_c \bar{j} \lambda = 0$
(\texttt{triplet\_cancellation}), and for pairs the identity
$j \overline{\lambda}^4 + \bar{j} \lambda^4 = 0$
(\texttt{pair\_cancellation}).

\begin{definition}[Hex edge direction]\label{def:hex_edge_dir}
\lean{hexEdgeDir}
\leanok
  The six possible edge directions in the hexagonal lattice,
  encoded as $\{0,1,2,3,4,5\}$ representing angles
  $0, \pi/3, 2\pi/3, \pi, -2\pi/3, -\pi/3$.
\end{definition}

\begin{definition}[Hex turn]\label{def:hex_turn}
\lean{hexTurn}
\leanok
\uses{def:hex_edge_dir}
  The turning angle at a vertex: the change in direction
  (in units of $\pi/3$) from one edge to the next.
\end{definition}

\begin{lemma}[Hex turn values at FALSE vertex]\label{lem:hex_turns_false}
\lean{hexTurn_false_w1_w2}
\leanok
\uses{def:hex_turn}
  At a FALSE vertex $v = (x,y,\text{false})$ with neighbors
  $w_1 = (x,y,\text{true})$, $w_2 = (x{+}1,y,\text{true})$,
  $w_3 = (x,y{+}1,\text{true})$:
  \begin{itemize}
  \item $\texttt{hexTurn}(w_1, v, w_2) = -1$ (turn $-\pi/3$)
  \item $\texttt{hexTurn}(w_1, v, w_3) = +1$ (turn $+\pi/3$)
  \item $\texttt{hexTurn}(w_2, v, w_1) = +1$ (turn $+\pi/3$)
  \item $\texttt{hexTurn}(w_2, v, w_3) = -1$ (turn $-\pi/3$)
  \item $\texttt{hexTurn}(w_3, v, w_1) = -1$ (turn $-\pi/3$)
  \item $\texttt{hexTurn}(w_3, v, w_2) = +1$ (turn $+\pi/3$)
  \end{itemize}
\end{lemma}
\begin{proof}
\leanok
  Direct computation by unfolding definitions.
\end{proof}

\begin{lemma}[Walk winding factoring]\label{lem:winding_factor}
\lean{walkWindingInt_cons_cons}
\leanok
\uses{def:hex_turn}
  For a walk $u \to v_1 \to \ldots$ of length $\geq 2$:
  $\mathrm{walkWindingInt}(\mathrm{cons}~h_1~(\mathrm{cons}~h_2~\mathrm{rest})) = \mathrm{hexTurn}(u, v_1, w_1) + \mathrm{walkWindingInt}(\mathrm{cons}~h_2~\mathrm{rest})$.
\end{lemma}
\begin{proof}\leanok By definitional unfolding (\texttt{rfl}).\end{proof}

\begin{lemma}[Full winding factoring]\label{lem:full_winding_factor}
\lean{fullWinding_cons_cons}
\leanok
\uses{lem:winding_factor}
  For a walk $u \to v_1 \to \ldots \to w_1$ of length $\geq 2$:
  $\mathrm{fullWinding}(\mathrm{cons}~h_1~q, w_{\mathrm{exit}}) = \mathrm{hexTurn}(u, v_1, q.\mathrm{getVert}~1) + \mathrm{fullWinding}(q, w_{\mathrm{exit}})$.
\end{lemma}
\begin{proof}\leanok
  Case split on $q$ (length $\geq 1$), unfold definitions, use \texttt{walkLastDir\_cons\_cons}
  and \texttt{walkWindingInt\_cons\_cons}.
\end{proof}

\begin{lemma}[Triplet winding property]\label{lem:triplet_winding}
\lean{triplet_winding_property}
\leanok
\uses{def:hex_turn, lem:full_winding_factor}
  For a walk $\gamma$ from $u$ to $w_1$ of length $\geq 1$, extended by one step through $v$:
  \[
    \mathrm{fullWinding}(\gamma \cdot (w_1 \to v), w_2) = \mathrm{fullWinding}(\gamma, v) + \mathrm{hexTurn}(w_1, v, w_2).
  \]
  The winding difference is a \emph{constant} that depends only on the local
  geometry at $v$, not on the specific walk $\gamma$. This is the key
  telescoping property needed for the triplet cancellation in the vertex relation.
\end{lemma}
\begin{proof}\leanok
  By induction on $\gamma$. Base case: direct computation. Inductive step: use
  \texttt{fullWinding\_cons\_cons} to factor out the first edge, apply IH to the tail.
\end{proof}

\begin{lemma}[Triplet cancellation (geometric)]\label{lem:triplet_cancel_geom}
\lean{triplet_cancellation_hex}
\leanok
\uses{lem:triplet_winding}
  With winding differences $-1$ and $+1$ (in units of $\pi/3$):
  $1 + x_c \cdot e^{-i\sigma(-\pi/3)} \cdot j + x_c \cdot e^{-i\sigma(\pi/3)} \cdot \bar{j} = 0$.
  This is equivalent to \texttt{triplet\_cancellation}.
\end{lemma}
\begin{proof}
\leanok
  Proved by converting to \texttt{triplet\_cancellation} via the
  identification $e^{i\sigma\pi/3} = \overline{\lambda}$ and
  $e^{-i\sigma\pi/3} = \lambda$.
\end{proof}

\section{Proof gaps and dependencies}\label{sec:gaps}

The main theorem (\S\ref{sec:main_thm}) depends on exactly
\textbf{two} independent sorry'd lemmas (reduced from three by
showing that the finite strip identity follows from the infinite one):

\begin{enumerate}
\item \textbf{Infinite strip identity}
  (\texttt{infinite\_strip\_identity} in
  \texttt{SAWRecurrenceProof.lean}):
  For the infinite strip $S_T$ with $T \ge 1$:
  \[
    1 = c_\alpha \cdot A_\infty(T, x_c) + x_c \cdot B(T, x_c).
  \]
  This is the $L \to \infty$ limit of the finite strip identity
  (with escape term $E \to 0$). Alternatively, it can be proved
  directly by the parafermionic argument on the infinite strip
  (no escape boundary). Used for the bridge recurrence
  $B(T) \le c_\alpha \cdot B(T{+}1)^2 + B(T{+}1)$ and hence for
  the lower bound $\mu \ge \sqrt{2+\sqrt{2}}$.
  \textbf{Also implies the finite strip identity}
  (\texttt{strip\_identity\_from\_infinite'} in
  \texttt{SAWParafermionicProof.lean}):
  since \texttt{PaperSAW\_B} $\hookrightarrow$
  \texttt{PaperBridge} (walks in the finite strip embed into infinite
  strip bridges), we get $B_{\text{paper}} \le x_c \cdot B \le 1$.
  This chain eliminates the finite strip identity as an independent sorry.

\item \textbf{Hammersley--Welsh decomposition}
  (\texttt{paper\_bridge\_decomp\_injection} in
  \texttt{SAWPaperChain.lean}):
  \[
    \sum_{n \leq N} c_n x^n \leq 2 \left(
    \sum_{S \subseteq \{1,\ldots,N\}}
    \prod_{T \in S} B_{T+1}(x)\right)^{\!2}.
  \]
  Purely combinatorial; independent of the observable.
  Key infrastructure (walk splitting, diagonal coordinate bounds,
  bridge length bounds, weight bounds) is proved.
  The remaining gap is the half-plane walk bridge decomposition
  algorithm and its injectivity proof.
  Required for $Z(x) < \infty$ when $x < x_c$ and hence for
  the upper bound $\mu \le \sqrt{2+\sqrt{2}}$.
\end{enumerate}

\subsection*{Proof sketch for the strip identity (Lemmas~1 and~2)}

For each interior vertex $v$ of the strip domain, the set of SAWs
from the starting mid-edge $a$ that reach one of $v
s three mid-edges
$p,q,r$ is partitioned into:
\begin{itemize}
\item \textbf{Triplets.} A walk $\gamma$ visiting only one mid-edge of
$v$ pairs with two extensions $\gamma_q, \gamma_r$ (adding one step
through $v$). The algebraic cancellation is
\texttt{triplet\_cancellation}:
$1 + x_c j \bar\lambda + x_c \bar j \lambda = 0$.
\item \textbf{Pairs.} A walk visiting all three mid-edges of $v$
pairs with a ``loop-reversed'' walk. The cancellation is
\texttt{pair\_cancellation}:
$j \bar\lambda^4 + \bar j \lambda^4 = 0$.
\end{itemize}
Summing over all interior vertices (discrete Stokes), interior
mid-edges cancel by direction antisymmetry, leaving only boundary
contributions:
$0 = -1 + c_\alpha A + B + c_\varepsilon E$.
Non-negativity of $A, E$ follows from \texttt{boundary\_cos\_pos}.

\subsection*{Proof sketch for the HW decomposition}

For each SAW $\gamma$ of length $n$ from the origin:
\begin{enumerate}
\item Split $\gamma$ at the first vertex of minimum diagCoord $d_{\min}$.
\item The prefix (reversed) and suffix are half-plane walks.
\item Each half-plane walk decomposes into bridges by induction on width:
  find the last vertex at max diagCoord, extract a bridge, recurse.
\item Bridge widths form a strictly decreasing subset of $\{1,\ldots,n\}$.
\item Walk length $\ge$ sum of bridge lengths, so
  $x^n \le \prod x^{\text{bridge length}}$.
\item The map $(\gamma, \text{choice}) \mapsto
  (\text{left bridges}, \text{right bridges})$ is injective.
\item Factor~$2$ accounts for two half-edge choices from the starting
  mid-edge.
\end{enumerate}

\item $Z(x) < +\infty$ for $0 < x < x_c$ (convergence below the critical point)
\end{enumerate}

\textbf{Fact 1} requires:
\begin{itemize}
\item The strip identity $1 = c_\alpha A + B + c_\varepsilon E$ (Lemma~2) --- \textbf{sorry}
\item The infinite strip identity $1 = c_\alpha A_{\inf} + x_c B_{\inf}$ --- \textbf{sorry}
\item The cutting argument $A_{T+1} - A_T \leq x_c B_{T+1}^2$ --- \textbf{proved}
\item The bridge recurrence $B_T \leq c_\alpha B_{T+1}^2 + B_{T+1}$ --- proved from above
\item The bridge lower bound $B_T \geq c/T$ --- proved from recurrence
\item $\sum_T B_T = +\infty$ since $B_T \geq c/T$ (harmonic divergence)
\end{itemize}

\textbf{Fact 2} requires:
\begin{itemize}
\item The Hammersley--Welsh counting inequality --- \textbf{sorry}
\item Bridge decay $B_T^x \leq (x/x_c)^T$ --- needs strip identity
\item Product convergence $\prod(1 + B_T^x) < \infty$ for $x < x_c$
\end{itemize}

\textbf{Root sorry count: 3} (strip identity, infinite strip identity,
HW decomposition). All other results are proved from these three.
